Function MyHex(ByVal Number)
	Dim Sign
	Const HexChars = "0123456789ABCDEF"
	Sign = Sgn(Number)
	Number = Fix(Abs(CDbl(number)))
	If Number = 0 Then
		MyHex = "0"
		Exit Function
	End If
	While Number > 0
		MyHex = Mid(HexChars, 1 + (Number - 16 * Fix(Number / 16)), 1) & MyHex
		Number = Fix(Number/16)
	WEnd
	If Sign = -1 Then MyHex = "-" & MyHex
End Function
